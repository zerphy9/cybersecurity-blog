#!/bin/bash  # эта строка "шебанг" системе какой интерпретатор нужно использовать

bash-скрипт - обычный текстовый файл с разрешением .sh, содержащий команды оболочки, которые выполняются по порядку
	важно сделать данный файл исполняемым, чтобы его команды выполнялись: chmod +x <имя_скрипта>.sh

синтаксис языка:
	указывание переменных и команд bash:
		указывание переменных осуществляется знаком $имя_переменной
		указывание команд bash осуществляется знаком $(команда_bash)
	передача аргументов:
		при необходимости передать значение скрипту, значение указывается после строки запуска скрипта: ./greet.sh Иван
		$0 - имя скрипта; $1, $2 - первый и второй переменные с аргументами, переданные при запуска скрипта

команды:
	echo "Привет!" - команда выведет "Привет!"
	date - выведет текущую дату и время в виде "Вс 03 авг 2025 20:03:49 MSK"
	name="Иван" - создается переменная
	# комментарий - так указываются комментарии

условия:
	синтаксис условий:
		if [ <условие> ]; then
			команда1
			команда2
		else
			команда3
		fi
	типы условий:
		[ "$name" = "Иван" ] - условие сравнения строк

цикл:
	синтаксис цикла for:
		for <переменная> in <список>; do
			команда
		done
	синтаксис цикла while:
		while [ <условие> ]; do
			команда
		done

python модуль subprocess и команды bash:
	subprocess - встроенный модуль в Python, который позволяет запускать команды Bash прямо из Python-кода
	возможности:
		запуск внешних команд и программ, ожидание процесса завершения
		получение вывода (stdout), ошибок (stderr), передача данных (stdin)
	класс:
		Popen - гибкий класс для запуска и управления процессом
	функции:
		run() - запускает команду, ждет её завершения и возвращает обьект с результатами
			параметры:
				args=[] - список значений, каждый по отдельности (прим. ['ls', '-l']
				text=True - чтобы работать с текстом, а не с байтами
				check=True - выбросит исключение, если команда завершится с ошибкой
				stdout/stderr - (необ. арг) куда направить вывод/ошибки
		call() - запуск процесса и ожидание завершения
		check_output() - запуск команды и получение ее вывода
